26. Remove Duplicates from Sorted Array

class Solution {
    public int removeDuplicates(int[] nums) {
        int k = 0;
        for(int i = 1; i < nums.length; i++){
            if(nums[i] > nums[k]){
                k++;
                int g;
                g=nums[k];
                nums[k] = nums[i];
                nums[i] = g;
            }
        }
    return k+1;
    }
}
